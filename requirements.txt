# The Mintlify Sentinel — Python dependencies
# Pin to exact versions for reproducible CI builds.
# Upgrade strategy: bump versions quarterly after testing.

# --- Core pipeline dependencies ---

# Pydantic v2: used by judge_config.py for docs.json schema validation.
# v2.x is required (judge_config.py uses model_validate(), model_config, etc.).
pydantic==2.7.4

# Jinja2: used by architect_render.py for MDX changelog template rendering.
jinja2==3.1.4

# PyYAML: allows load_openapi() to parse YAML-format OpenAPI specs in addition
# to JSON. Many real-world specs are written in YAML; without this dependency
# the diff engine would reject them with an "INVALID JSON" error.
pyyaml==6.0.2

# --- Streamlit UI ---

# Browser interface for the Sentinel pipeline.
streamlit==1.35.0

# --- Testing ---

# pytest: test runner for the tests/ suite.
pytest==8.2.2

# pytest-cov: coverage reporting plugin.
pytest-cov==5.0.0
